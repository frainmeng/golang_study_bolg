## 变量定义
  + var variableName type
  + var vname1, vname2, vname3 type
  + var variableName type = value
  + var vname1, vname2, vname3 type= v1, v2, v3
  + var vname1, vname2, vname3 = v1, v2, v3
  + vname1, vname2, vname3 := v1, v2, v3//只能用在函数内部
